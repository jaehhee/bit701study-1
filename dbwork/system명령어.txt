--연습용 계정 생성- 한줄 실행 ctrl+enter
create user c##scott identified by a1234;
--오라클 최신버전은 계정앞에 c## 을 붙여서 생성해야 한다
--그런데 c##  을 안붙이고 생성하려면 아래 코드 추가후 생성하면 된다
alter session set "_ORACLE_SCRIPT"=true
--위의 c##scott 제거
drop user c##scott;
--scott 라는 계정 생성하기
create user scott identified by a1234;
--기본권한주기
grant connect,resource to scott;

--계정 생성후 테이블 space  오류시 아래 sql문 넣어준다
alter user scott default tablespace users quota unlimited on users;
